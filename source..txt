local function fetchScript(url)
    local cacheBuster = string.format("?v=%d&r=%d&t=%d", tick() * 1000, math.random(100000, 999999), os.time())
    local success, result = pcall(function()
        return game:HttpGet(url .. cacheBuster, true)
    end)
    return success and result or nil
end

-- [[ REAL-TIME SYSTEM NOTIFICATION FUNCTION ]]
local function sendNotification(title, message)
    pcall(function()
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = title,
            Text = message,
            Duration = 5
        })
    end)
end

-- [[ GET GREETING BASED ON REAL-TIME ]]
local function getRealTimeGreeting()
    local currentHour = tonumber(os.date("%H"))
    if currentHour >= 0 and currentHour < 12 then
        return "Good morning!"
    elseif currentHour >= 12 and currentHour < 18 then
        return "Good afternoon!"
    else
        return "Good evening!"
    end
end
